66 // ***************************************************************************
67 // Struct: structure_mapping_info //DX20191212 //DX20201218 - updated
68 // ***************************************************************************
69 // Stores the structural similarity between two structures
70 // (i.e., the representative structure and a matched structure)
71 // The basis transformation, rotation, origin shift, and atom mapping
72 // information is also stored (new as of 20201218).
73 struct structure_mapping_info {
74  // quantitative similarity/cost functions
75  double misfit; // structural misfit (=1-(1-lattice_deviation)(1-coordinate_displacement)(1-failure)) (see Burzlaff)
76  double lattice_deviation; // lattice deviation, captures differences between lattices
77  double coordinate_displacement; // coordinate displacement; captures differences between atom positions (relatively close together)
78  double failure; // figure of failure; captures differences between atom positions (significantly far apart)
79  bool is_magnetic_misfit; // boolean indicating if a magnetic system and using magnetic as misfit
80  double magnetic_misfit; // magnetic misfit (DX inspired by Burzlaff's misfit; =1-(1-magnetic_displacement)(1-magnetic_failure))
81  double magnetic_displacement; // magnetic displacement; captures differences between magnetic moment magnitude (and angle for non-collinear)
82  double magnetic_failure; // magnetic failure; captures spin flip differences
83  // transformation info
84  double rescale_factor; // rescaling factor for duplicate structure
85  xmatrix<double> rotation; // rotation for duplicate structure (rotate into representative reference frame)
86  xmatrix<double> basis_transformation; // basis transformation for duplicate structure (convert to lattice similar to reference)
87  xvector<double> origin_shift; // shift origin for duplicate structure (shift to common origin with representative structure)
88  // mapping info
89  vector<uint> atom_map; // atom index map: position in vector corresponds to reference index, value corresponds to duplicate structure index
90  vector<uint> basis_map; // atom type map: position in vector corresponds to reference index, value corresponds to duplicate structure type
91  vector<double> distances_mapped; // distances between mapped atoms between reference and duplciate structures
92  vector<xvector<double> > vectors_mapped; // mapping vectors between mapped atoms between reference and duplicate structures
93 };
